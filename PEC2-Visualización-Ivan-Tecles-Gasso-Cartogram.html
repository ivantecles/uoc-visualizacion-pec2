library(sf)
## Linking to GEOS 3.10.2, GDAL 3.4.2, PROJ 8.2.1; sf_use_s2() is TRUE
library(ggplot2)
library(viridis)
## Loading required package: viridisLite
epsg_code <- 3857

# African countries
data(wrld_simpl)
africa = wrld_simpl[wrld_simpl$REGION==2,]

# Construct a cartogram using the population in 2005
africa_sf <- st_as_sf(africa) %>% 
  st_transform(epsg_code)

africa_cartogram <- cartogram_cont(africa_sf, "POP2005", itermax=5)
## Mean size error for iteration 1: 5.79263926219508
## Mean size error for iteration 2: 4.94711300669559
## Mean size error for iteration 3: 4.32566127362514
## Mean size error for iteration 4: 3.84933064483894
## Mean size error for iteration 5: 3.45947755254362
# Cartogram
ggplot() +
  geom_sf(data = africa_cartogram, aes(fill = POP2005/1000000)) +
  scale_size(
    range = c(1, 9), 
    guide = guide_legend(
      direction = "horizontal", 
      nrow = 1, 
      label.position = "bottom")
  ) +
  scale_fill_viridis(name="Population (M)", breaks=c(1, 20, 40, 60, 80, 100, 120, 140, 141)) +
  guides(fill = guide_legend(title = "")) +
  labs(title = "Poblacin por Pas en Africa 2005", size = "") +
  theme(legend.position = "bottom")
